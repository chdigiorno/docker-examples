# Sin un entrypoint, luego de levantar la aplicación, habría que hacer todo esto a mano:

docker-compose exec web bash  # Entro al container
cd ejemplo_de_django_en_docker_compose

# Correr migraciones
python manage.py makemigrations
python manage.py migrate

# Crear superuser de Django
python manage.py createsuperuser

# Comprobar que existe en la base de datos
Manera 1:
	exit  # Salgo del container para entrar al de la db
	docker-compose exec db bash  # Entro al container de la db
	psql -U postgres  # Entro a postgres (duh)
	select * from auth_user;  # Acá debería aparecer el usuario que creé
Manera 2:
	apt-get update
	apt-get install postgresql-client  # Instalamos el cliente de postgres para entrar desde este contenedor
	psql -h db -U postgres
	select * from auth_user;

# Y recién acá corremos el runserver...
